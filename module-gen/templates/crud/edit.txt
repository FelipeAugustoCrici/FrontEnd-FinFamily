import { useNavigate, useParams } from 'react-router-dom';
import { {ModelUpper}Form } from './components/{ModelUpper}Form';
import { useUpdate{ModelUpper} } from './hooks/useUpdate{ModelUpper}';
import { use{ModelUpper} } from './hooks/use{ModelUpper}';

export function {ModelUpper}Edit() {
  const navigate = useNavigate();
  const { id } = useParams<{ id: string }>();
  const update{ModelUpper} = useUpdate{ModelUpper}();

  const { data: item, isLoading } = use{ModelUpper}(id);

  if (isLoading) {
    return (
      <div className="flex justify-center py-20">
        <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-primary-500" />
      </div>
    );
  }

  if (!item) {
    return (
      <div className="max-w-4xl mx-auto py-20 text-center">
        <p className="text-primary-500">{ModelVisible} não encontrado</p>
      </div>
    );
  }

  return (
    <{ModelUpper}Form
      initialData={item}
      onSubmit={(data) =>
        update{ModelUpper}.mutate(
          { id: id!, data },
          {
            onSuccess: () => navigate('/{ModelLower}'),
          },
        )
      }
      isLoading={update{ModelUpper}.isPending}
    />
  );
}
