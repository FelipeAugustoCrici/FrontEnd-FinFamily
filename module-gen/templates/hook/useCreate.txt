import { useMutation, useQueryClient } from '@tanstack/react-query';
import { useToast } from '@/hooks/useToast';
import { {ModelLower}Service } from '../services/{ModelLower}.service';
import { {ModelUpper}FormData } from '../components/{ModelUpper}Form';

export function useCreate{ModelUpper}() {
  const queryClient = useQueryClient();
  const { showToast } = useToast();

  return useMutation({
    mutationFn: (data: {ModelUpper}FormData) => {ModelLower}Service.create(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['{ModelLower}s'] });
      showToast({
        title: 'Sucesso',
        description: '{ModelUpper} criado com sucesso!',
        variant: 'success',
      });
    },
    onError: () => {
      showToast({
        title: 'Erro',
        description: 'Erro ao criar {ModelLower}',
        variant: 'error',
      });
    },
  });
}
