import { useMutation, useQueryClient } from '@tanstack/react-query';
import { useToast } from '@/hooks/useToast';
import { {ModelLower}Service } from '../services/{ModelLower}.service';

export function useDelete{ModelUpper}() {
  const queryClient = useQueryClient();
  const { showToast } = useToast();

  return useMutation({
    mutationFn: (id: string) => {ModelLower}Service.delete(id),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['{ModelLower}s'] });
      showToast({
        title: 'Sucesso',
        description: '{ModelUpper} excluído com sucesso!',
        variant: 'success',
      });
    },
    onError: () => {
      showToast({
        title: 'Erro',
        description: 'Erro ao excluir {ModelLower}',
        variant: 'error',
      });
    },
  });
}
